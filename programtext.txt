#include <stdio.h>
int main()
{
    int i, f, l, m, n, t, a[100];
    printf("How many numbers ?\n");
    scanf("%d", &n);
    printf("enter %d integers\n", n);
    for (int i = 0; i < n; i++)
    {
        scanf("%d", &a[i]);
    }

    printf("which value to find?\n");
    scanf("%d", &t);
    f = 0;
    l = n - 1;
    m = (f + l) / 2;
    while (f <= l)
    {
        if (a[m] < t)
            f = m + 1;
        else if (a[m] == t)
        {
            printf("The number %d found at location %d \n", t, m + 1);
            break;
        }
        else
            l = m - 1;

        m = (f + l) / 2;

        if (f > l)
        {
            printf("The number %d not found in list ", t);
        }
    }

    return 0;
}
